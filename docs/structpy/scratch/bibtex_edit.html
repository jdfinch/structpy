def process_bibtex(bibtex_string):
    for line in bibtex_string.split('\n'):
        i = line.find('title={')
        if i != -1:
            y = None
            for j in range(i, len(line)):
                if line[j] == '}':
                    y = j
                    break
            x = i+len('title={')
            line = line[:x] + '{' + line[x:y] + '}' + line[y:]
        print(line)

if __name__ == '__main__':
    process_bibtex(bibtex)
